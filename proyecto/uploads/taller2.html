#Seleccion de columnas para el modelo kmeans
misDatos=data[['Annual Income (k$)','Spending Score (1-100)']]
scaler=StandardScaler()
# manejo de la misma escala en los datos, para este caso no es necesario porque manejan rangos parecidos pero de igual forma se aplic
datosEscala=scaler.fit_transform(misDatos)

#Uso del mtodo del codo para la obtencion de un numero de cluster adecuado
inercia=[]
for k in range (1,10):
    kmeans=KMeans(n_clusters=k,random_state =42).fit(datosEscala)
    inercia.append(kmeans.inertia_)

plt.figure(figsize =[8,5])
plt.title('Mtodo del codo')
plt.xlabel('Nmero de clsters')
plt.ylabel('Inercia')
plt.plot ( list(range(1,10)), inercia, marker='o')
plt.show()
